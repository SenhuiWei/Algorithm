    class Solution {
    public:
    int numberOfSubarrays(vector<int>& nums, int k) {
    // use the prefix sum
    unordered_map <int, int> hash;

    int ans = 0, tot = 0;
    hash[0] = 1;
    for (auto x : nums) {
    if (x & 1) x = 1;
    else x = 0;
    tot += x;
    // add the number of prefixes that add up to tot - k
    ans += hash[tot - k]; 
    hash[tot] ++;
    }

    return ans;
    }
    };

    \end{verbatim}
